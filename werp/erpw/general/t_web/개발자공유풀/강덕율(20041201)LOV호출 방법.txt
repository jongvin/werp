-- LOV 호충방법 --
function btnPreView_onClick()
{
	var lrArgs = new C_Dictionary(); // 입력인자를 넘기기 위한 객체(주의:넘길 인자가 없어도 반드시 생성해야함)
	var lrRet = null; // LOV에서 선택한 결과를 넘겨받기 위한 변수선언
	
	lrArgs.set("PROJ_CODE", "A00001");
	
	lrRet = C_LOV("TestLov1", lrArgs); // C_LOV(LOV명, 입력인자)
	
	txtProjCode.value = lrRet.get("PROJ_CODE"); // 넘겨받은 값을 사용
	txtProjName.value = lrRet.get("PROJ_NAME");
}
