	부서및 현장 찾기 window.showModalDialog의 argment 사용방법.
	첫번째 argment 는 프로그램 id 즉 html 
        예) var is_dept_search_addr = gs_home + '/comm_search/z_dept_find_1h.html'
	두번째 argment 는 넘겨줄 변수 (주의: 반드시 gs_home을 반드시 넘겨줘야한다)
	     예) gs_home + "@" + 넘겨줄 data
	     예) gs_home + "@" + data
	 세번째 argment는 화면의 위치 정보 
	
	결과값은 result에 셋팅 되며
	   array = result.split("@"); 이문장으로 array에 결과값들이 들어간다.
 전체 사용예제는 다음과 같이 사용하면 됩니다.
 
	
	var result;
	var array = new Array();
	result = window.showModalDialog(is_dept_search_addr,gs_home + "@" + data,"dialogWidth:500px;dialogHeight:500px");
	if(result == "") return;

	array = result.split("@");
